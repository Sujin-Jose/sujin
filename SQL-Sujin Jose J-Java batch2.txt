
create database vehicleShowroom;
use vehicleShowroom;


create table category(
category_id int primary key auto_increment,
category_type varchar(30));


create table customer(customer_id int primary key auto_increment ,
customer_name varchar(30),
email varchar(50),
phone_number int);


create table vendor(vendor_id int primary key auto_increment,
vendor_name varchar(30),
email varchar(30),
phone_number int);
drop table product;


create table product(product_id	int primary key	auto_increment,
product_name varchar(35),
category_id int,vendor_id int, price int, quantity int, 
product_description varchar(100), 
foreign key (category_id) references category(category_id),
foreign key (vendor_id) references vendor(vendor_id));
